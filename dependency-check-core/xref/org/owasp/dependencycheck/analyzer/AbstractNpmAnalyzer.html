49  /**
50   * An abstract NPM analyzer that contains common methods for concrete
51   * implementations.
52   *
53   * @author Steve Springett
54   */
55  @ThreadSafe
56  public abstract class AbstractNpmAnalyzer extends AbstractFileTypeAnalyzer {
57  
58      /**
59       * The logger.
60       */
61      private static final Logger LOGGER = LoggerFactory.getLogger(AbstractNpmAnalyzer.class);
62  
63      /**
64       * A descriptor for the type of dependencies processed or added by this
65       * analyzer.
66       */
67      public static final String NPM_DEPENDENCY_ECOSYSTEM = "npm";
68      /**
69       * The file name to scan.
70       */
71      private static final String PACKAGE_JSON = "package.json";
72  
73      /**
74       * Determines if the file can be analyzed by the analyzer.
75       *
76       * @param pathname the path to the file
77       * @return true if the file can be analyzed by the given analyzer; otherwise
78       * false
79       */
80      @Override
81      public boolean accept(File pathname) {
82          boolean accept = super.accept(pathname);
83          if (accept) {
84              try {
85                  accept = shouldProcess(pathname);
86              } catch (AnalysisException ex) {
87                  throw new UnexpectedAnalysisException(ex.getMessage(), ex.getCause());
88              }
89          }
90          return accept;
91      }
92  
93      /**
94       * Determines if the path contains "/node_modules/" or "/bower_components/"
95       * (i.e. it is a child module). This analyzer does not scan child modules.
96       *
97       * @param pathname the path to test
98       * @return <code>true</code> if the path does not contain "/node_modules/"
99       * or "/bower_components/"
100      * @throws AnalysisException thrown if the canonical path cannot be obtained
101      * from the given file
102      */
103     protected boolean shouldProcess(File pathname) throws AnalysisException {
104         try {
105             // Do not scan the node_modules (or bower_components) directory
106             final String canonicalPath = pathname.getCanonicalPath();
107             if (canonicalPath.contains(File.separator + "node_modules" + File.separator)
108                     || canonicalPath.contains(File.separator + "bower_components" + File.separator)) {
109                 LOGGER.debug("Skipping analysis of node/bower module: {}", canonicalPath);
110                 return false;
111             }
112         } catch (IOException ex) {
113             throw new AnalysisException("Unable to process dependency", ex);
114         }
115         return true;
116     }

118     /**
119      * Construct a dependency object.
120      *
121      * @param dependency the parent dependency
122      * @param name the name of the dependency to create
123      * @param version the version of the dependency to create
124      * @param scope the scope of the dependency being created
125      * @return the generated dependency
126      */
127     protected Dependency createDependency(Dependency dependency, String name, String version, String scope) {
128         final Dependency nodeModule = new Dependency(new File(dependency.getActualFile() + "?" + name), true);
129         nodeModule.setEcosystem(NPM_DEPENDENCY_ECOSYSTEM);
130         //this is virtual - the sha1 is purely for the hyperlink in the final html report
131         nodeModule.setSha1sum(Checksum.getSHA1Checksum(String.format("%s:%s", name, version)));
132         nodeModule.setSha256sum(Checksum.getSHA256Checksum(String.format("%s:%s", name, version)));
133         nodeModule.setMd5sum(Checksum.getMD5Checksum(String.format("%s:%s", name, version)));
134         nodeModule.addEvidence(EvidenceType.PRODUCT, "package.json", "name", name, Confidence.HIGHEST);
135         nodeModule.addEvidence(EvidenceType.VENDOR, "package.json", "name", name, Confidence.HIGH);
136         if (!StringUtils.isBlank(version)) {
137             nodeModule.addEvidence(EvidenceType.VERSION, "package.json", "version", version, Confidence.HIGHEST);
138             nodeModule.setVersion(version);
139         }
140         if (dependency.getName() != null) {
141             nodeModule.addProjectReference(dependency.getName() + ": " + scope);
142         } else {
143             nodeModule.addProjectReference(dependency.getDisplayFileName() + ": " + scope);
144         }
145         nodeModule.setName(name);
146 
147         //TODO  - we can likely create a valid CPE as a low confidence guess using cpe:2.3:a:[name]_project:[name]:[version]
148         //(and add a targetSw of npm/node)
149         Identifier id;
150         try {
151             final PackageURL purl = PackageURLBuilder.aPackageURL().withType(StandardTypes.NPM)
152                     .withName(name).withVersion(version).build();
153             id = new PurlIdentifier(purl, Confidence.HIGHEST);
154         } catch (MalformedPackageURLException ex) {
155             LOGGER.debug("Unable to generate Purl - using a generic identifier instead " + ex.getMessage());
156             id = new GenericIdentifier(String.format("npm:%s@%s", dependency.getName(), version), Confidence.HIGHEST);
157         }
158         nodeModule.addSoftwareIdentifier(id);
159         return nodeModule;
160     }

162     /**
163      * Processes a part of package.json (as defined by JsonArray) and update the
164      * specified dependency with relevant info.
165      *
166      * @param engine the dependency-check engine
167      * @param dependency the Dependency to update
168      * @param jsonArray the jsonArray to parse
169      * @param depType the dependency type
170      */
171     protected void processPackage(Engine engine, Dependency dependency, JsonArray jsonArray, String depType) {
172         final JsonObjectBuilder builder = Json.createObjectBuilder();
173         jsonArray.getValuesAs(JsonString.class).forEach((str) -> builder.add(str.toString(), ""));
174         final JsonObject jsonObject = builder.build();
175         processPackage(engine, dependency, jsonObject, depType);
176     }
177 
178     /**
179      * Processes a part of package.json (as defined by JsonObject) and update
180      * the specified dependency with relevant info.
181      *
182      * @param engine the dependency-check engine
183      * @param dependency the Dependency to update
184      * @param jsonObject the jsonObject to parse
185      * @param depType the dependency type
186      */
187     protected void processPackage(Engine engine, Dependency dependency, JsonObject jsonObject, String depType) {
188         for (int i = 0; i < jsonObject.size(); i++) {
189             jsonObject.forEach((name, value) -> {
190                 String version = "";
191                 if (value != null && value.getValueType() == ValueType.STRING) {
192                     version = ((JsonString) value).getString();
193                 }
194                 final Dependency existing = findDependency(engine, name, version);
195                 if (existing == null) {
196                     final Dependency nodeModule = createDependency(dependency, name, version, depType);
197                     engine.addDependency(nodeModule);
198                 } else {
199                     existing.addProjectReference(dependency.getName() + ": " + depType);
200                 }
201             });
202         }
203     }

205     /**
206      * Adds information to an evidence collection from the node json
207      * configuration.
208      *
209      * @param dep the dependency to add the evidence
210      * @param t the type of evidence to add
211      * @param json information from node.js
212      * @return the actual string set into evidence
213      * @param key the key to obtain the data from the json information
214      */
215     private static String addToEvidence(Dependency dep, EvidenceType t, JsonObject json, String key) {
216         String evidenceStr = null;
217         if (json.containsKey(key)) {
218             final JsonValue value = json.get(key);
219             if (value instanceof JsonString) {
220                 evidenceStr = ((JsonString) value).getString();
221                 dep.addEvidence(t, PACKAGE_JSON, key, evidenceStr, Confidence.HIGHEST);
222             } else if (value instanceof JsonObject) {
223                 final JsonObject jsonObject = (JsonObject) value;
224                 for (final Map.Entry<String, JsonValue> entry : jsonObject.entrySet()) {
225                     final String property = entry.getKey();
226                     final JsonValue subValue = entry.getValue();
227                     if (subValue instanceof JsonString) {
228                         evidenceStr = ((JsonString) subValue).getString();
229                         dep.addEvidence(t, PACKAGE_JSON,
230                                 String.format("%s.%s", key, property),
231                                 evidenceStr,
232                                 Confidence.HIGHEST);
233                     } else {
234                         LOGGER.warn("JSON sub-value not string as expected: {}", subValue);
235                     }
236                 }
237             } else {
238                 LOGGER.warn("JSON value not string or JSON object as expected: {}", value);
239             }
240         }
241         return evidenceStr;
242     }

244     /**
245      * Locates the dependency from the list of dependencies that have been
246      * scanned by the engine.
247      *
248      * @param engine the dependency-check engine
249      * @param name the name of the dependency to find
250      * @param version the version of the dependency to find
251      * @return the identified dependency; otherwise null
252      */
253     protected Dependency findDependency(Engine engine, String name, String version) {
254         for (Dependency d : engine.getDependencies()) {
255             if (NPM_DEPENDENCY_ECOSYSTEM.equals(d.getEcosystem()) && name.equals(d.getName()) && version != null && d.getVersion() != null) {
256                 final String dependencyVersion = d.getVersion();
257                 if (DependencyBundlingAnalyzer.npmVersionsMatch(version, dependencyVersion)) {
258                     return d;
259                 }
260             }
261         }
262         return null;
263     }

265     /**
266      * Collects evidence from the given JSON for the associated dependency.
267      *
268      * @param json the JSON that contains the evidence to collect
269      * @param dependency the dependency to add the evidence too
270      */
271     public void gatherEvidence(final JsonObject json, Dependency dependency) {
272         String displayName = null;
273         if (json.containsKey("name")) {
274             final Object value = json.get("name");
275             if (value instanceof JsonString) {
276                 final String valueString = ((JsonString) value).getString();
277                 displayName = valueString;
278                 dependency.setName(valueString);
279                 dependency.setPackagePath(valueString);
280                 dependency.addEvidence(EvidenceType.PRODUCT, PACKAGE_JSON, "name", valueString, Confidence.HIGHEST);
281                 dependency.addEvidence(EvidenceType.VENDOR, PACKAGE_JSON, "name", valueString, Confidence.HIGH);
282             } else {
283                 LOGGER.warn("JSON value not string as expected: {}", value);
284             }
285         }
286         //TODO - if we start doing CPE analysis on node - we need to exclude description as it creates too many FP
287         final String desc = addToEvidence(dependency, EvidenceType.PRODUCT, json, "description");
288         dependency.setDescription(desc);
289         final String vendor = addToEvidence(dependency, EvidenceType.VENDOR, json, "author");
290         final String version = addToEvidence(dependency, EvidenceType.VERSION, json, "version");
291         if (version != null) {
292             displayName = String.format("%s:%s", displayName, version);
293             dependency.setVersion(version);
294 
295             Identifier id;
296             try {
297                 final PackageURL purl = PackageURLBuilder.aPackageURL()
298                         .withType(StandardTypes.NPM).withName(dependency.getName()).withVersion(version).build();
299                 id = new PurlIdentifier(purl, Confidence.HIGHEST);
300             } catch (MalformedPackageURLException ex) {
301                 LOGGER.debug("Unable to generate Purl - using a generic identifier instead " + ex.getMessage());
302                 id = new GenericIdentifier(String.format("npm:%s:%s", dependency.getName(), version), Confidence.HIGHEST);
303             }
304             dependency.addSoftwareIdentifier(id);
305         }
306         if (displayName != null) {
307             dependency.setDisplayFileName(displayName);
308             dependency.setPackagePath(displayName);
309         } else {
310             LOGGER.warn("Unable to determine package name or version for {}", dependency.getActualFilePath());
311             if (vendor != null && !vendor.isEmpty()) {
312                 dependency.setDisplayFileName(String.format("%s package.json", vendor));
313             }
314         }
315         // Adds the license if defined in package.json
316         if (json.containsKey("license")) {
317             final Object value = json.get("license");
318             if (value instanceof JsonString) {
319                 dependency.setLicense(json.getString("license"));
320             } else if (value instanceof JsonArray) {
321                 final JsonArray array = (JsonArray) value;
322                 final StringBuilder sb = new StringBuilder();
323                 boolean addComma = false;
324                 for (int x = 0; x < array.size(); x++) {
325                     if (!array.isNull(x)) {
326                         if (addComma) {
327                             sb.append(", ");
328                         } else {
329                             addComma = true;
330                         }
331                         if (ValueType.STRING == array.get(x).getValueType()) {
332                             sb.append(array.getString(x));
333                         } else {
334                             final JsonObject lo = array.getJsonObject(x);
335                             if (lo.containsKey("type") && !lo.isNull("type")
336                                     && lo.containsKey("url") && !lo.isNull("url")) {
337                                 final String license = String.format("%s (%s)", lo.getString("type"), lo.getString("url"));
338                                 sb.append(license);
339                             } else if (lo.containsKey("type") && !lo.isNull("type")) {
340                                 sb.append(lo.getString("type"));
341                             } else if (lo.containsKey("url") && !lo.isNull("url")) {
342                                 sb.append(lo.getString("url"));
343                             }
344                         }
345                     }
346                 }
347                 dependency.setLicense(sb.toString());
348             } else {
349                 dependency.setLicense(json.getJsonObject("license").getString("type"));
350             }
351         }
352     }
